word_hist = (names.flatMap(lambda line: line.split(" ")) # Convert to a list of words (i.e. separated by space) \
.map(lambda w: (w, 1))# For each word, emit (word, 1)
.reduceByKey(lambda x, y: x + y) # Add up the 1s to get the number of occurrences
.sortBy(lambda r: r[1], ascending=False).take(100)) # Sort on occurrences descending
